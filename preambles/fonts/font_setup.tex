% For proper font
\usepackage[no-math]{fontspec}

% Times New Roman within text
\setmainfont{Times New Roman}

% Times New Roman in equations
\usepackage{unicode-math}
\setmathfont{TeX Gyre Termes Math}

    "0023, % Hashtag
    "0025, % Percent sign
    "0026, % Ampersand
    "002C, % Comma
    "003B, % Semicolon
    \infty,
    \partial
}]
\setmathfont[range=\mathup/{num, latin, Latin}]{Times New Roman}
\setmathfont[range=\mathit/{latin, Latin}]{Times New Roman Italic}
\setmathfont[range=\mathup/{greek, Greek}]{Times New Roman}
\setmathfont[range=\mathit/{greek, Greek}]{Times New Roman}
