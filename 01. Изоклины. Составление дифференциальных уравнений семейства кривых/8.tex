Уравнение изоклины: $k + y = \br{x + k}^3$, $y = \br{x + k}^3 - x$, $x = \sqrt[3]{y + k} + k$. Изоклинами будут кубические параболы.

\begin{figure}[!h]
    \centering
    \begin{tikzpicture}
        \pgfplotsset{
            Graph/.style = {
                samples = 100,
                restrict y to domain = -5 : 5,
            },
            DirectionGraph/.style = {
                samples = 80,
                restrict y to domain = -5 : 5,
                quiver = {scale arrows = 0.075},
            },
            DirectionGraph1/.style = {
                DirectionGraph,
                quiver = {
                    u = {dot_x(x, y) / sqrt(dot_x(x, y) ^ 2 + dot_y(x, y) ^ 2)},
                    v = {dot_y(x, y) / sqrt(dot_x(x, y) ^ 2 + dot_y(x, y) ^ 2)}
                }
            },
            DirectionGraph2/.style = {
                DirectionGraph,
                quiver = {
                    u = {-dot_x(x, y) / sqrt(dot_x(x, y) ^ 2 + dot_y(x, y) ^ 2)},
                    v = {-dot_y(x, y) / sqrt(dot_x(x, y) ^ 2 + dot_y(x, y) ^ 2)}
                }
            },
        }
        \begin{axis}[
            trig format plots=rad,
            width = 12cm,
            height = 10cm,
            ymin = -4.2,
            ymax = 4.2,
            xmin = -4.2,
            xmax = 4.2, 
            xtick = {-10, ..., 10},
            ytick = {-10, ..., 10},
            minor tick num = 3,
            declare function = {
                f(\k, \x) = (x - \k)^3 - \k;
                dot_x(\x, \y) = 1;
                dot_y(\x, \y) = \s * \k;
            }]

            \foreach \k in {0, 0.25, ..., 4} {
                \foreach \s in {-1, 1} {
                    \addplot[Graph, gray, domain = -4.5 : 4.5]
                    {f(\s * \k, x)};
                    \addplot[DirectionGraph1, domain = -4.5 : 4.5]
                    {f(\s * \k, x)};
                    \addplot[DirectionGraph2, domain = -4.5 : 4.5]
                    {f(\s * \k, x)};
                }
            }
        \end{axis}
    \end{tikzpicture}
\end{figure}
