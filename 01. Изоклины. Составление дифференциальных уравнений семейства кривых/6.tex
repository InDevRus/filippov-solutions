$xy' = 2y$. $y' = \dfrac {2y} {x}$.

Итак, $f\br{x; y} = \dfrac {2y} {x}$. Уравнение изоклины: $\dfrac {2y} {x} = k$. $y = \dfrac {k} {2} x$. Таким образом, изоклинами будут прямые. В точке $x = 0$, $y = 0$ поведение касательной нерегулярно. На левом рисунке изображены изоклины, на правом -- решения уравнений.

В частности, из построенных изоклин видно, что среди решений есть прямая $y_{0} = 0$. Действительно, всякая изоклина -- это прямая с угловым коэффициентом $\dfrac {k} {2}$, так что, если положить $y'\br{x} = f\br{x; y} = k$ равной также $0$, то получится прямая $y_{0} = 0$.

    \centering
    \begin{tikzpicture}
        \pgfplotsset{
            Graph/.style = {
                samples = 20,
                restrict y to domain = -5 : 5,
            },
            DirectionGraph/.style = {
                samples = 20,
                quiver = {scale arrows = 0.075},
            },
            DirectionGraph1/.style = {
                DirectionGraph,
                quiver = {
                    u = {dot_x(x, y) / sqrt(dot_x(x, y) ^ 2 + dot_y(x, y) ^ 2)},
                    v = {dot_y(x, y) / sqrt(dot_x(x, y) ^ 2 + dot_y(x, y) ^ 2)}
                }
            },
            DirectionGraph2/.style = {
                DirectionGraph,
                quiver = {
                    u = {-dot_x(x, y) / sqrt(dot_x(x, y) ^ 2 + dot_y(x, y) ^ 2)},
                    v = {-dot_y(x, y) / sqrt(dot_x(x, y) ^ 2 + dot_y(x, y) ^ 2)}
                }
            },
        }
        \begin{axis}[
            width = 7.5cm,
            height = 10cm,
            ymin = -4.2,
            ymax = 4.2,
            xmin = -4.2,
            xmax = 4.2, 
            xtick = {-10, ..., 10},
            ytick = {-10, ..., 10},
            minor tick num = 3,
            declare function = {
                f(\k, \x) = 0.5 * \k * x;
                dot_x(\x, \y) = x;
                dot_y(\x, \y) = 2 * y;
            }]
            
            \foreach \k in {0, 0.25, ..., 2, 2.5, 3, 4, 5} {
                \foreach \s in {-1, 1} {
                    \addplot[Graph, gray, domain = -4.5 : 4.5]
                    {f(\s * \k, x)};
                    \addplot[DirectionGraph1, domain = 0.4 : 4]
                    {f(\s * \k, x)};
                    \addplot[DirectionGraph2, domain = 0.4 : 4] 
                    {f(\s * \k, x)};
                    \addplot[DirectionGraph1, domain = -4 : -0.4]
                    {f(\s * \k, x)};
                    \addplot[DirectionGraph2, domain = -4 : -0.4]
                    {f(\s * \k, x)};
                }
            }
        \end{axis}
    \end{tikzpicture}
\end{minipage}
\begin{minipage}{.5\textwidth}
    \centering
    \begin{tikzpicture}
        \pgfplotsset{
            Graph/.style = {
                samples = 20,
                restrict y to domain = -5 : 5,
                opacity = 0.5,
            },
            PreciseGraph/.style = {
                Graph,
                samples = 100,
                opacity = 1,
            },
            DirectionGraph/.style = {
                samples = 20,
                quiver = {scale arrows = 0.075},
                opacity = 0.5,
            },
            DirectionGraph1/.style = {
                DirectionGraph,
                quiver = {
                    u = {dot_x(x, y) / sqrt(dot_x(x, y) ^ 2 + dot_y(x, y) ^ 2)},
                    v = {dot_y(x, y) / sqrt(dot_x(x, y) ^ 2 + dot_y(x, y) ^ 2)}
                }
            },
            DirectionGraph2/.style = {
                DirectionGraph,
                quiver = {
                    u = {-dot_x(x, y) / sqrt(dot_x(x, y) ^ 2 + dot_y(x, y) ^ 2)},
                    v = {-dot_y(x, y) / sqrt(dot_x(x, y) ^ 2 + dot_y(x, y) ^ 2)}
                }
            },
        }
        \begin{axis}[
            width = 7.5cm,
            height = 10cm,
            ymin = -4.2,
            ymax = 4.2,
            xmin = -4.2,
            xmax = 4.2, 
            xtick = {-10, ..., 10},
            ytick = {-10, ..., 10},
            minor tick num = 3,
            declare function = {
                f(\k, \x) = 0.5 * \k * x;
                dot_x(\x, \y) = x;
                dot_y(\x, \y) = 2 * y;
                g(\C, \x) = \C * x * x;
            }]
            
            \foreach \k in {0, 0.25, ..., 2, 2.5, 3, 4, 5} {
                \foreach \s in {-1, 1} {
                    \addplot[Graph, gray, domain = -4.5 : 4.5]
                    {f(\s * \k, x)};
                    \addplot[DirectionGraph1, domain = 0.4 : 4]
                    {f(\s * \k, x)};
                    \addplot[DirectionGraph2, domain = 0.4 : 4] 
                    {f(\s * \k, x)};
                    \addplot[DirectionGraph1, domain = -4 : -0.4]
                    {f(\s * \k, x)};
                    \addplot[DirectionGraph2, domain = -4 : -0.4]
                    {f(\s * \k, x)};
                }
            }
            
            \foreach \C in {0, 0.1, 0.25, 0.5, 1, 2, 3} {
                \foreach \s in {-1, 1} {
                    \addplot[PreciseGraph, domain = -4.5 : 4.5]
                    {g(\s * \C, x)};
                }
            }
        \end{axis}
    \end{tikzpicture}
